//A Helpful way to visualize code:
//   Prewritten code to make your programs come to life
public class Iterations
{
public static void main (String[] args)
{
//Creating a ball at x = 0 and y = 0
Iterate("ball", new Position(0,0));
//Loops can make many balls appear with little code!
for (int i = -5; i <= 5; i++)
{
if (i != 0)
{
Iterate("ball", new Position(i,2));
}
}
ClearIterations();
//Nested Loops can make patterns, too!
for (int i = 0; i < 6; i++)
{
for (int j = 0; j < i; j++)
{
Iterate("box", new Position(j,i));
}
}
}
}
