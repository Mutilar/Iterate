//Writing more efficiently:
//   Making repetitve tasks more compact
public class Loops
{
public static void main (String[] args)
{
int value = 10;
//While an argument is true, run whatever is inside the loop
while (value != 0)
{
value--;
}
System.out.println(value);
while (value < 10) 
{
value += 2;
}
System.out.println(value);
//For loops compress while loops into one neat package
for (int new_value = -4; new_value < 3; new_value++)
{
System.out.println(new_value);
}
}
}
